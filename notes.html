Querying Data

  • JOINs (INNER, LEFT, RIGHT)

-- Example query
SELECT customers.name, COUNT(orders.id) as order_count
FROM customers
LEFT JOIN orders ON customers.id = orders.customer_id
GROUP BY customers.name
ORDER BY order_count DESC;
